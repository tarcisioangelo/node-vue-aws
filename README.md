### To set up Node.js on your Linux instance
- curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.39.7/install.sh | bash
- source ~/.bashrc
- nvm install --lts
- node -e "console.log('Running Node.js ' + process.version)"
